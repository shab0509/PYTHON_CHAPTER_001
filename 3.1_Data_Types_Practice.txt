# Four types of number system available:
# Decimal Number System

a= 10
print(a)
print(a,type(a))

# Binary Number System

a = 0b010101 
print(a)
print(type(a))

a = 0B0000101011
print(a)
print(type(a))

b=21
c = bin(b)
print(c)

# Octal Number System

a= 0o11
print(a)
print(type(a))

b= 9
c = oct(b)
print(c)

# Hexadecimal Number System
a= 0x1A1
print(a)
print(type(a))

b= 417 
c = hex(b)
print(c)


a= 0X1A11
print(a)
print(type(a))
